import { Books, Votes } from '../models';
 
 
export default class votesController {
  /**
       *  method to upVote/downVote a book
       * @method addVotes
       * @param {object} req
       * @param {object} res
       * @return {json}
       */
  addupVote(req, res) {
    this.req = req;
    this.res = res;
    return Votes.create({
      userId: req.params.userId,
      bookId: req.params.bookId,
      upVotes: 1,
 
    });
  }
 
  updateUpvotes(req, res) {
    this.req = req;
    this.res = res;
 
    const bookId = parseInt(req.params.bookId, 10);
    return Books
      .findById(bookId)
 
      .then((book) => {
        if (!book) return { statusCode: 404, msg: 'Book not found' };
        return book;
      })
      .then((book) => {
        book.increment('upVotes');
        return book;
      })
      .then((book) => {
        book.reload();
        return book;
      })
 
      .then(book => res.status(201).send({ msg: `Upvotes increased to ${book.upVotes}` }))
      .catch(error => res.status(500).send(error));
  }
 
 
  addDownVote(req, res) {
    this.req = req;
    this.res = res;
    return Votes.create({
      userId: req.params.userId,
      bookId: req.params.bookId,
      downVotes: 1,
 
    });
  }
 
  updateDownVote(req, res) {
    this.req = req;
    this.res = res;
    const bookId = parseInt(req.params.bookId, 10);
    return Books
      .findById(bookId)
 
      .then((books) => {
        if (!books) {
          return res.status(404).send({
            message: 'Book Not Found',
          });
        }
        return Books
          .update({
            bookId: Books.bookId,
            bookName: Books.bookName,
            description: Books.description,
            author: Books.author,
            quantity: Books.quantity,
            publishYear: Books.publishYear,
            downVotes: Books.increment({ downVotes: 1 }),
 
 
          })
          .then(() => res.status(200).send(Books)) // Send back the updated book.
          .catch(error => res.status(400).send(error));
      })
      .catch(error => res.status(400).send(error));
  }
}
